System Calls

  1. open, read, write, close System Calls

  2. open

    int open(const char *pathname, int flags, mode_t mode);

    1. flags, O_RDONLY, O_WRONLY, or O_RDWR

    2. flags, O_APPEND, O_CREAT, O_NONBLOCK (does not work on regular files), O_TRUNC

    3. mode, S_IRWXU, S_IRUSR, S_IRWXG, S_IRGRP, S_IRWXO, S_IROTH

  3. read

    ssize_t read(int fd, void *buf, size_t count);

    1. 0 → EOF,

    2. Non-blocking read

      char buf[BUFSIZ];
ssize_t nr;
start:
  nr = read (fd, buf, BUFSIZ);
  if (nr == -1) {
    if (errno == EINTR)
      goto start;
    if (errno == EAGAIN)
      /* resubmit later */
    else
      /* error */
  }

  4. write

    ssize_t write (int fd, const void *buf, size_t count);

    1. partial write example

      ssize_t ret, nr;
while (len != 0 && (ret = write (fd, buf, len)) != 0) {
  if (ret == -1) {
    if (errno == EINTR)
      continue;
    perror ("write");
  break;
  }
  len -= ret;
  buf += ret;
}

  1. fsync

    int fsync (int fd); // Flushes both data and metadata (timestamps, other attributes..)
int fsyncdata (int fd); // Flushes on data, not guarantee of metadata

    1. O_SYNC flag in open can also be used to force synchronization

    2. O_DSYNC - Defined by POSIX, same as fsyncdata()

    3. O_RSYNC - Defined by POSIX, same as fsync()

  2. lseek

    #include<unistd.h>
#include<sys/types.h>
off_t lseek (int fd, off_t pos, int origin);

    1. SEEK_CUR The current file position of fd is set to its current value plus pos, which can be negative, zero, or positive. A pos of zero returns the current file position value.

    2. SEEK_END The current file position of fd is set to the current length of the file plus pos, which can be negative, zero, or positive. A pos of zero sets the offset to the end of the file.

    3. SEEK_SET The current file position of fd is set to pos. A pos of zero sets the offset to the beginning of the file.

  3. pipes, O_NONBLOCK, fcntl(fd, F_SETFL, fcntl(fd, F_GETFL) | O_NONBLOCK);
